	title = ron_diplomatic.4.t
	desc = ron_diplomatic.4.d
	picture = GFX_report_event_preww2_country40
	fire_only_once = yes
	
    trigger = {
		tag = GRE
		date > 1934.1.1
		date < 1935.1.1
		has_country_flag = balkan_pact_mutual_defense
			AND = {
				ROM = { has_country_flag = balkan_pact_yes }
				YUG = { has_country_flag = balkan_pact_yes }
				TUR = { has_country_flag = balkan_pact_yes }
			}
	}
	
	mean_time_to_happen = {
		days = 7
	}
	
	option = {
		#Form The Pact
		name = ron_diplomatic.4.a
		
		ai_chance = {
			factor = 10
		}
		
		GRE = {
		give_guarantee = ROM
		give_guarantee = TUR
		give_guarantee = YUG
		}
		
		ROM = {
		give_guarantee = GRE
		give_guarantee = TUR
		give_guarantee = YUG
		}
		
		YUG = {
		give_guarantee = ROM

		}
		
		TUR = {
		give_guarantee = ROM

		}
		
		add_political_power = 100
