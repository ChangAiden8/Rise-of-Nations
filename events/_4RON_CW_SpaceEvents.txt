# First intercontinental ballistic missile 
news_event = {
id = roncw_spacenews.1
title = roncw_spacenews.1.t
desc = roncw_spacenews.1.d
picture = GFX_news_event_space_first_icbm

major = yes

trigger = {
	NOT = { has_global_flag = space_first_icbm }
	date > 1957.8.21
	date < 2000.1.1
	has_start_date < 1960.01.01
	SOV = { has_tech = RON_ICBM1 }
	SOV = { has_tech = rocket_engines }
	NOT = { SOV = { has_idea = nuclear_budget1 } }
	country_exists = SOV
}

immediate = {
     set_global_flag = space_first_icbm
}

mean_time_to_happen = {
        days = 7
    }

option = {
     name = roncw_spacenews.1.a
	 
	 trigger = { has_government = communism }
	 
     }
	 
option = {
     name = roncw_spacenews.1.b
	 
	 trigger = { NOT = { has_government = communism } }
	 
     }
}
