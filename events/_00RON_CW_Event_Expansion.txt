country_event = {
	id = CW_Expansion_country.10
	title = CW_Expansion_country.10.t
	desc = CW_Expansion_country.10.d
	picture = GFX_report_event_CW_Expansion_country10
	fire_only_once = yes
	trigger = {
		date > 1946.2.24
		date < 1947.1.1
		ARG = {
			is_subject = no
		}
		tag = ARG
	}
	option = {
		name = CW_Expansion_country.10.a
		ai_chance = {
			factor = 10
		}
		if = {
			limit = {
				NOT = {
					has_government = fascism
				}
			}
			set_politics = {
				ruling_party = fascism
				elections_allowed = no
			}
			add_popularity = {
				ideology = fascism
				popularity = 0.5
			}
		}
		recruit_character = ARG_juan_domingo_peron
		promote_character = ARG_juan_domingo_peron
		set_party_name = { ideology = fascism  long_name = ARG_partido_peronista_long name = ARG_partido_peronista }
		news_event = {
			id = CW_Expansion_news.10
			hours = 6
		}
	}
	option = {
		name = CW_Expansion_country.10.b
		
		trigger = {
		NOT = { has_government = fascism }
		}
		
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		
		random_list = {
			50 = {
				start_civil_war = {
					ruling_party = ROOT
					ideology = fascism
					size = 0.5
				}
				random_other_country = {
					limit = {
						original_tag = ARG
						has_government = fascism
					}
					recruit_character = ARG_juan_domingo_peron
					promote_character = ARG_juan_domingo_peron
					set_party_name = { ideology = fascism  long_name = ARG_partido_peronista_long name = ARG_partido_peronista }
				}
			}
			50 = {
				add_popularity = {
					ideology = fascism
					popularity = -0.25
				}
			}
		}
			
	}
}
