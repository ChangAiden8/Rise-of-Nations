	is_triggered_only = yes
	option = {
		name = RON_Russian_Alt_History.1.a
		ai_chance = {
			factor = 10
		}
		
		start_civil_war = {
			ruling_party = neutrality
			ideology = communism
			size = 0.5
		}
		
		hidden_effect = {
			create_country_leader = {
				name = "Andrey Vlasov"
				picture = "gfx/leaders/WHR/Andrey_Vlasov.dds"
				ideology = militarism
				traits = {
					dictator
				}
			}
		}
		if = {
			limit = {
				has_country_flag = RON_Stalin_Escapes
			}
		random_other_country = {
			limit = {
				original_tag = SOV
				has_government = communism
			}
			create_country_leader = {
				name = "Joseph Stalin"
				desc = "POLITICS_JOSEPH_STALIN_DESC"
				picture = "Portrait_Soviet_Joseph_Stalin.dds"
				ideology = stalinism
				traits = {
					man_of_steel
				}
			}
		}
		}
		if = {
			limit = {
				has_country_flag = RON_Stalin_Killed
			}
			random_other_country = {
				limit = {
					original_tag = SOV
					has_government = communism
				}
				retire_ideology_leader = communism
				create_country_leader = {
					name = "Lavrentiy Beria"
					picture = "gfx/leaders/SOV/Portrait_SOV_Beria.dds"
					ideology = stalinism
				}
			}
		}
		RON_increase_conscription_law = yes
		RON_increase_economy_law = yes
		
		set_global_flag = RON_Second_Russian_Civil_War
		
		news_event = {
			id = RON_Russian_Alt_History_News.1
			hours = 6
		}
			
	}
}
