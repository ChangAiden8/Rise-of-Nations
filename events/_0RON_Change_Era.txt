			}
			RON_give_russian_winter_event = yes
		}

		if = {
			limit = {
				threat > 24
				threat < 50
			}
			add_named_threat = {
				threat = -25
				name = "Era Change"
			}
		}

		if = {
			limit = {
				threat > 49
				threat < 75
			}
			add_named_threat = {
				threat = -50
				name = "Era Change"
			}
		}

		if = {
			limit = {
				threat > 74
				threat < 99
			}
			add_named_threat = {
				threat = -75
				name = "Era Change"
			}
		}

		if = {
			limit = {
				threat > 99
			}
			add_named_threat = {
				threat = -100
				name = "Era Change"
			}
		}

	}
