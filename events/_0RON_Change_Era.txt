		name = change_era.1.a

			factor = 1000
		}

		custom_effect_tooltip = RON_loading_next_era_tt
		hidden_effect = {
			RON_change_era_ok_effects = yes
		}
		every_other_country = {
			limit = {
				#is_ai = yes
				RON_must_change_era = yes
			}
			country_event = { id = change_era.2 }
		}
		country_event = { id = power_status.1 }
		set_global_flag = power_status_in_progress
		clr_global_flag = change_era_in_progress

		if = {
			limit = {
				RON_must_give_russia_winter_idea = yes
				NOT = { has_global_flag = RON_weather_report_in_progress }
			}
			RON_give_russian_winter_event = yes
		}

		if = {
			limit = {
				threat > 24
				threat < 50
			}
			add_named_threat = {
				threat = -25
				name = "Era Change"
			}
		}

		if = {
			limit = {
				threat > 49
				threat < 75
			}
			add_named_threat = {
				threat = -50
				name = "Era Change"
			}
		}

		if = {
			limit = {
				threat > 74
				threat < 99
			}
			add_named_threat = {
				threat = -75
				name = "Era Change"
			}
		}

		if = {
			limit = {
				threat > 99
			}
			add_named_threat = {
				threat = -100
				name = "Era Change"
			}
		}
		
		#Facu's Event System Properly Implemented
		#MUST be implemented here, it can't be implemented in the OK/Immediate Effects cause Facu's system is meant to be only #executed by 1 country
		#WW1 Leaders
		if = {
			limit = {
				date > 1909.12.31
				date < 1919.1.1
			}
			country_event = { id = change_era.6 hours = 1 }
		}
		#Interwar
		if = {
			limit = {
				date > 1918.12.31
				date < 1933.1.1
			}
			country_event = { id = change_era.7 hours = 1 }
		}
		#WW2
		if = {
			limit = {
				date > 1932.12.31
				date < 1949.1.1
			}
			country_event = { id = change_era.8 hours = 1 }
			country_event = { id = change_era.12 hours = 1 }
		}
		#Early Cold War
		if = {
			limit = {
				date > 1948.12.31
				date < 1980.1.1
			}
			country_event = { id = change_era.9 hours = 1 }
			country_event = { id = change_era.13 hours = 1 }
		}
		#Late Cold War
		if = {
			limit = {
				date > 1979.12.31
				date < 2000.1.1
			}
			country_event = { id = change_era.10 hours = 1 }
		}
		#Modern Day
		if = {
			limit = {
				date > 1999.12.31
			}
			country_event = { id = change_era.11 hours = 1 }
			country_event = { id = change_era.14 hours = 1 }
		}

		every_country = {
			mark_focus_tree_layout_dirty = yes
		}

	}
}
