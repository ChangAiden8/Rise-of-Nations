country_event = {
	id = change_era.1
	title = change_era.1.t
	desc = change_era.1.d
	picture = "[GFX_RON_Welcome_Event_By_Date]"
	#trigger = {
	#	is_ai = no
	#	RON_must_change_era = yes
	#}
	#mean_time_to_happen = {
	#	days = 30
	#}
	is_triggered_only = yes
	#hidden = yes
	immediate = {
		hidden_effect = {
			RON_change_era_immediate_effects = yes
		}
	}
	option = {
		name = change_era.1.a

		ai_chance = {
			factor = 1000
		}

		custom_effect_tooltip = RON_loading_next_era_tt
		hidden_effect = {
			RON_change_era_ok_effects = yes
		}
		every_other_country = {
			limit = {
				#is_ai = yes
				RON_must_change_era = yes
			}
			country_event = { id = change_era.2 }
		}
		country_event = { id = power_status.1 }
		set_global_flag = power_status_in_progress
		clr_global_flag = change_era_in_progress

		if = {
			limit = {
				RON_must_give_russia_winter_idea = yes
				NOT = { has_global_flag = RON_weather_report_in_progress }
			}
			RON_give_russian_winter_event = yes
		}

		if = {
			limit = {
				threat > 24
				threat < 50
			}
			add_named_threat = {
				threat = -25
				name = "Era Change"
			}
		}

		if = {
			limit = {
				threat > 49
				threat < 75
			}
			add_named_threat = {
				threat = -50
				name = "Era Change"
			}
		}

		if = {
			limit = {
				threat > 74
				threat < 99
			}
			add_named_threat = {
				threat = -75
				name = "Era Change"
			}
		}

		if = {
			limit = {
				threat > 99
			}
			add_named_threat = {
				threat = -100
				name = "Era Change"
			}
		}

	}
}
