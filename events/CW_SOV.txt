#Stalin dies
country_event = {
	id= cw_sov.12
	title = cw_sov.12.t
	desc = cw_sov.12.d

	is_triggered_only = yes
	fire_only_once = yes
	trigger = {
		tag = SOV
		has_country_leader = {
			character = joseph_stalin
			ruling_only = yes
		}
		date > 1953.3.5
		date < 1953.4.5
	}
	option = {
		name = cw_sov.12.a
		if = {
			limit = {
				has_country_leader = {
					character = joseph_stalin
					ruling_only = yes
				}
			}
			kill_country_leader = yes
			
		}
		set_variable = { malenkov_influence = 30}
		set_variable = { beria_influence = 80}
		set_variable = { khrushchev_influence = 35 }
		set_variable = { zhukov_influence = 10}
		set_variable = { bulganin_influence = 15 }
		
	}
}
