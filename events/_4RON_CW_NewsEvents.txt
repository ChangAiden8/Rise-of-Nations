# Churchill delivers Iron Curtain speech
news_event = {
id = roncw_news.1
title = roncw_news.1.t
desc = roncw_news.1.d
picture = GFX_news_event_iron_curtain_speech

major = yes

trigger = {
	NOT = { has_global_flag = iron_curtain_speech }
	date > 1946.3.4
	date < 1946.4.4
	SOV = { has_government = communism }
	NOT = { ENG = { has_government = communism } }
	country_exists = SOV
	country_exists = ENG
}

immediate = {
	set_global_flag = iron_curtain_speech
}



option = {
	name = roncw_news.1.a
	}
}
