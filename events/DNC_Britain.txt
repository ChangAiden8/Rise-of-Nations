#Ultimatum to FRA to restore House Platagenet (FRA)
country_event = {
	id = dnc_eng.7
	title = dnc_eng.7.t
	desc = dnc_eng.7.d

	is_triggered_only = yes

	option = {
		#accept
		name = dnc_eng.7.a
		if = {
			limit = {
				tag = FRA
			}
			VIF = {
				every_state = {
					limit = {
						is_claimed_by = ENG
						is_controlled_by = FRA
					}
					transfer_state = PREV
				}
			}
			ENG = {
				puppet = VIF
			}
		}
	}

	option = {
		#refuse
		name = dnc_eng.7.b
		ENG = {
			create_wargoal = {
				target = FRA
				type = take_claimed_state
			}
		}
	}
}
