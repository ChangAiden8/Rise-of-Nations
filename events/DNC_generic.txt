##Disband 8 Nation Alliance
country_event = {
	id = dnctree.19
	picture = GFX_report_event_military_planning
	title = dnctree.19.title
	desc = dnctree.19.desc
	fire_only_once = yes
	hidden = yes
	trigger = {
		tag = FRA
		is_faction_leader = yes
		NOT = { has_war_with = BOX }
		OR = {
		is_in_faction_with = GER
		is_in_faction_with = AUH
		is_in_faction_with = ITA
		is_in_faction_with = JAP
		}
		date > 1901.9.7
		date < 1910.1.1
	}
	mean_time_to_happen = {
		days = 7
	}
	option = {
		name = dnctree.19.a
		ai_chance = {
			factor = 1000
		}
		
		every_other_country = {
		limit = { is_in_faction_with = FRA }
 		leave_faction = yes
		}
		
		FRA = { dismantle_faction = yes }
		
		}
}
