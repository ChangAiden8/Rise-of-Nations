add_namespace = weather_events
########Universal Event############
## RUSSIAN WINTER
country_event = {
	id = weather_events.1
	title = weather_events.1.t
	desc = weather_events.1.d
	picture = GFX_report_event_soviet_tanks_snow
	trigger = {
		OR = {
		tag = SOV
		tag = RUE
		tag = WHR
		tag = RUS
		}
		OR = {
			AND = {
				is_russian_winter = yes
				has_country_flag = russian_summer
			}
			AND = {
				is_russian_winter = yes
				NOT = { has_country_flag = russian_winter }
			}
			AND = {
				is_russian_winter = no
				has_country_flag = russian_winter
			}
			AND = {
				is_russian_winter = no
				NOT = { has_country_flag = russian_summer }
			}
		}
	}
	mean_time_to_happen = {
		days = 2
	}
	hidden = yes
	immediate = {
		#Army
		if = {
			limit = {
				is_russian_winter = yes
			}
			set_country_flag = russian_winter
			clr_country_flag = russian_summer
			
		}
		if = {
			limit = {
				is_russian_winter = no
			}
			set_country_flag = russian_summer
			clr_country_flag = russian_winter
		}
		
	}
	option = {
		name = "Ok"
		#Weather
		if = { limit = { has_country_flag = russian_winter } add_ideas = russian_winter }
		if = { limit = { has_country_flag = russian_summer } remove_ideas = russian_winter }

	}
}
