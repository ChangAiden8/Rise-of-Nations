# Paris Universal Exhibition
news_event = {
	id = worldnews.5
	title = worldnews.5.t
	desc = worldnews.5.d
	picture = GFX_Expo
	major = yes

	trigger = {
		NOT = { has_global_flag = paris_universal_1937 }
		date > 1937.5.25
		date < 1937.6.25
		country_exists = FRA
		FRA = { surrender_progress < 0.10 }
		FRA = { has_civil_war = no }
		NOT = { any_country = { is_major = yes has_war_with = FRA } }
	}

	immediate = {
		set_global_flag = paris_universal_1937
	}

	mean_time_to_happen = {
		days = 7
	}
	option = { name = worldnews.5.a }
immediate = { log = "Event: [ROOT.GetName] [GetDateString] file:France_Mod_Events.txt line:103" }}
