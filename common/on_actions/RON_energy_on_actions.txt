# 1 coal = 0.6 gas
# 1 oil = 2 coal

on_actions = {
	on_weekly = {
		effect = {
			ROOT = {
				
				set_variable = {
					num_civs = 0
				}
				set_variable = {
					num_arms = 0
				}
				set_variable = {
					num_ref = 0
				}
				every_controlled_state = {
					add_to_variable = {
						ROOT.num_civs = building_level@industrial_complex
					}
					add_to_variable = {
						ROOT.num_arms = building_level@arms_factory
					}
					add_to_variable = {
						ROOT.num_ref = building_level@synthetic_refinery
					
					}
				}
				calculate_shortages = yes
				if = {
					limit = {
						NOT = {
							has_dynamic_modifier = {
								modifier = RON_energy_consumption
							}
						}
					}
					add_dynamic_modifier = {
						modifier = RON_energy_consumption
					}
				}
				
			}
		}
		
	}
}
