on_actions = {

on_government_change = {
		effect = {			
			if = {
				
			limit = { has_government = fascism }
			PER = { add_ideas = PER_king_reza_fascism }	
			}

			if = {
			limit = { has_government = communism }
			PER = { add_ideas = PER_king_reza_communism }
			}

			if = {
			limit = { has_government = democratic }
			PER = { add_ideas = PER_king_reza_democratic }
			}		
		}
	}
}
