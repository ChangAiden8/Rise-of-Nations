			OR = {

				original_tag = SPR
			}
		}
		available = {
			always = yes
		}
		ai_will_do = {
			factor = 10
			modifier = {
				factor = 0
				NOT = {tag = MOB}
				FROM = {
					is_owned_by = MOB
				}
				OR = {
					has_country_flag = support_MOR
					tag = MOR
				}
			}
		}
		state_target = yes
		icon = revolt
		target_trigger = {
			FROM = {
				has_state_flag = MOR_resistance
			}
		}
		visible = {
			FROM = {
				has_state_flag = MOR_resistance
			}
			ROOT = {
				OR = {  
					original_tag = MOR
					has_country_flag = joined_moroccan_pacification_campaign
				}
			}
		}
		modifier = {
			resistance_target = 0.1
		}
		days_remove = 150
		cost = 20
		remove_effect = {
			random_list = {
				33 = {
					modifier = {
						add = 33
						tag = MOR
					}
					FROM = {
						add_resistance = 5
					}
					
				}

					modifier = {
						add = -33
						tag = MOR
					}
					FROM = {
