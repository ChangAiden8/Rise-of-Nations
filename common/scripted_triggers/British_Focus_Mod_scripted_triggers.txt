Second_World_War_happening = { #This won't really work on an "anything goes" play-through, it pretty much has to be historical
	custom_trigger_tooltip = {
		tooltip = Second_World_War_happening
	}
	hidden_trigger = {
		ENG = {
			has_government = democratic
			OR = {
				
				#1st Condition
				AND = {
					GER = { has_government = fascism }
					ITA = { has_government = fascism }
					JAP = { has_government = fascism }
					has_war_with = GER
					has_war_with = ITA
					has_war_with = JAP
					
				}
				
				#2nd Condition
				AND = {
					GER = { has_government = fascism }
					ITA = { has_government = fascism }
					has_war_with = GER
					has_war_with = ITA
					NOT = { has_war_with = JAP }
					JAP = {
						OR = {
							is_in_faction_with = ENG
						}
						NOT = {
							has_war_with = event_target:WTT_current_china_leader
							has_government = fascism
						}
					}
				}
				
				#3rd Condition
				AND = {
					GER = { has_government = fascism }
					JAP = { has_government = fascism }
					has_war_with = GER
					has_war_with = JAP
					ITA = {
						has_government = fascism
						is_in_faction_with = GER
						has_capitulated = yes
					}
				}
				
				#4th Condition
				AND = {
					ITA = { has_government = fascism }
					JAP = { has_government = fascism }
					has_war_with = ITA
					has_war_with = JAP
					GER = {
						has_government = fascism
						is_in_faction_with = ITA
						has_capitulated = yes
					}
				}
				
				#5th Condition
				AND = {
					GER = { has_government = fascism }
					has_war_with = GER
					NOT = { has_war_with = JAP }
					JAP = {
						OR = {
							is_in_faction_with = ENG
						}
						NOT = {
							has_war_with = event_target:WTT_current_china_leader
							has_government = fascism
						}
					}
					ITA = {
						has_government = fascism
						is_in_faction_with = GER
						has_capitulated = yes
					}
				}
				
				#6th Condition
				AND = {
					ITA = { has_government = fascism }
					has_war_with = ITA
					JAP = {
						has_government = democratic
						OR = {
							is_in_faction_with = ENG
						}
					}
					GER = {
						has_government = fascism
						is_in_faction_with = ITA
						has_capitulated = yes
					}
				}
				
				#7th Condition
				AND = {
					JAP = { has_government = fascism }
					has_war_with = JAP
					NOT = {
						has_war_with = GER
						has_war_with = ITA
					}
					ITA = {
						OR = {
							NOT = {
								has_government = fascism
							}
							exists = no
						}
					}
					OR = {
						GER = {
							OR = {
								NOT = {
									has_government = fascism
								}
								exists = no
							}
						}
						AND = {
							WGR = { exists = yes }
							DDR = { exists = yes }
						}
					}
				}
			}
		}
		Second_World_War_Allied_Victory = no
	}
}
