join_league_of_nations_effect = {
	if = {
		limit = {
			NOT = { has_country_flag = member_of_league_of_nations }
			NOT = { 
				tag = UNA
				tag = LON
			}
		}
		set_country_flag = member_of_league_of_nations
		add_to_array = { global.members_of_league_of_nation_array = THIS.id }
		add_ideas = league_of_nations_member_idea
		if = {
			limit = {
				has_global_flag = league_of_nations_focus_1_completed
			}
			add_ideas = league_of_nations_focus_idea_1
		}
		if = {
			limit = {
				has_global_flag = league_of_nations_focus_2_completed
			}
			add_ideas = league_of_nations_focus_idea_2
		}
		if = {
			limit = {
				has_global_flag = league_of_nations_focus_3_completed
			}
			add_ideas = league_of_nations_focus_idea_3
		}
		if = {
			limit = {
				any_other_country = {
					check_variable = { num_of_active_league_of_nations_trade_sanctions > 0 }
				}
			}
			every_other_country = {
				limit = {
					check_variable = { num_of_active_league_of_nations_trade_sanctions > 0 }
				}
				if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 1 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_1
					}
				}
				else_if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 2 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_2
					}
				}
				else_if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 3 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_3
					}
				}
				else_if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 4 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_4
					}
				}
				else_if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 5 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_5
					}
				}
				else_if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 6 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_6
					}
				}
				else_if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 7 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_7
					}
				}
				else_if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 8 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_8
					}
				}
				else_if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 9 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_9
					}
				}
				else_if = {
					limit = {
						check_variable = { num_of_active_league_of_nations_trade_sanctions = 10 }
					}
					reverse_add_opinion_modifier = {
						target = PREV
						modifier = league_of_nations_trade_sanction_modifier_10
					}
				}
			}
		}
		if = {
			limit = {
				has_global_flag = league_of_nations_map_mode_active
			}
			save_cosmetic_tag_before_league_of_nations_map_mode = yes
			set_cosmetic_tag = LEAGUE_OF_NATIONS_MEMBER_CT
		}
	}
}

leave_league_of_nations_effect = {
	if = {
		limit = {
			has_country_flag = member_of_league_of_nations
		}
		clr_country_flag = member_of_league_of_nations
		diplomatic_relation = {
			country = LON
			relation = guarantee
			active = no
		}
		remove_from_array = { global.members_of_league_of_nation_array = THIS.id }
		remove_ideas = league_of_nations_member_idea
		if = {
			limit = {
				has_idea = league_of_nations_focus_idea_1
			}
			remove_ideas = league_of_nations_focus_idea_1
		}
		if = {
			limit = {
				has_idea = league_of_nations_focus_idea_2
			}
			remove_ideas = league_of_nations_focus_idea_2
		}
		if = {
			limit = {
				has_idea = league_of_nations_focus_idea_3
			}
			remove_ideas = league_of_nations_focus_idea_3
		}
		if = {
			limit = {
				has_country_flag = voted_in_noconfidence_vote
			}
			clr_country_flag = voted_in_noconfidence_vote
			subtract_from_variable = { global.num_of_league_of_nations_noconfidence_votes = 1 }
		}
		if = {
			limit = {
				has_country_flag = league_of_nations_voted_yes
			}
			clr_country_flag = league_of_nations_voted_yes
			subtract_from_variable = { global.num_of_yes_votes_in_league_of_nations_voting = 1 }
			if = {
				limit = {
					OR = {
						has_country_flag = leader_of_league_of_nations
						has_country_flag = was_leader_of_league_of_nations
					}
				}
				subtract_from_variable = { global.num_of_yes_votes_in_league_of_nations_voting = 2 }
			}
			if = {
				limit = {
					has_country_flag = league_of_nations_voted_sanction_type_trade
				}
				clr_country_flag = league_of_nations_voted_sanction_type_trade
				subtract_from_variable = { global.num_of_trade_sanction_votes_in_league_of_nations_voting = 1 }
			}
			if = {
				limit = {
					has_country_flag = league_of_nations_voted_sanction_type_war
				}
				clr_country_flag = league_of_nations_voted_sanction_type_war
				subtract_from_variable = { global.num_of_war_votes_in_league_of_nations_voting = 1 }
			}
			if = {
				limit = {
					has_country_flag = league_of_nations_voted_sanction_type_kick_out_of_league_of_nations
				}
				clr_country_flag = league_of_nations_voted_sanction_type_kick_out_of_league_of_nations
				subtract_from_variable = { global.num_of_kick_out_of_league_of_nations_votes_in_league_of_nations_voting = 1 }
			}
			if = {
				limit = {
					has_country_flag = league_of_nations_voted_sanction_type_form_coalition
				}
				clr_country_flag = league_of_nations_voted_sanction_type_form_coalition
				subtract_from_variable = { global.num_of_form_coalition_votes_in_league_of_nations_voting = 1 }
			}
		}
		if = {
			limit = {
				has_country_flag = league_of_nations_voted_no
			}
			clr_country_flag = league_of_nations_voted_no
			subtract_from_variable = { global.num_of_no_votes_in_league_of_nations_voting = 1 }
			if = {
				limit = {
					OR = {
						has_country_flag = leader_of_league_of_nations
						has_country_flag = was_leader_of_league_of_nations
					}
				}
				subtract_from_variable = { global.num_of_no_votes_in_league_of_nations_voting = 2 }
			}
		}
		if = {
			limit = {
				OR = {
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_1
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_2
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_3
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_4
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_5
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_6
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_7
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_8
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_9
					has_opinion_modifier = league_of_nations_trade_sanction_modifier_10
				}
			}
			country_event = { id = league_of_nations_events.20 }
		}
		if = {
			limit = {
				has_country_flag = leader_of_league_of_nations
			}
			clr_country_flag = leader_of_league_of_nations
			clear_global_event_target = league_of_nations_leader
			remove_ideas = league_of_nations_leader_idea
		}
		if = {
			limit = {
				has_global_flag = league_of_nations_map_mode_active
			}
			save_cosmetic_tag_before_league_of_nations_map_mode = yes
			set_cosmetic_tag = NOT_LEAGUE_OF_NATIONS_MEMBER_CT
		}
		if = {
			limit = {
				has_variable = elected_country_for_lon_leader
			}
			var:elected_country_for_lon_leader = {
				subtract_from_variable = { num_of_votes_for_lon_election = 1 }
			}
			clear_variable = elected_country_for_lon_leader
		}
		if = {
			limit = {
				any_other_country = {
					has_variable = elected_country_for_lon_leader
					check_variable = { elected_country_for_lon_leader = PREV.id }
				}
			}
			every_other_country = {
				limit = {
					has_variable = elected_country_for_lon_leader
					check_variable = { elected_country_for_lon_leader = PREV.id }
				}
				PREV = {
					subtract_from_variable = { num_of_votes_for_lon_election = 1 }
				}
				clear_variable = elected_country_for_lon_leader
			}
		}
	}
}
