

auh_bosnian_support_small_increase = {
	add_to_variable = {bosnian_support_percent = 5}
	auh_calculate_bosnian_icon_frame = yes
}

auh_bosnian_support_medium_increase = {
	add_to_variable = {bosnian_support_percent = 10}
	auh_calculate_bosnian_icon_frame = yes
}

auh_bosnian_support_medium_decrease = {
	subtract_from_variable = {bosnian_support_percent = 10}
	auh_calculate_bosnian_icon_frame = yes

}

auh_bosnian_support_small_decrease = {
	subtract_from_variable = {bosnian_support_percent = 5}
	auh_calculate_bosnian_icon_frame = yes 

}

auh_calculate_bosnian_icon_frame = {
	set_temp_variable = { temp_bos_support = bosnian_support_percent}
	divide_temp_variable = {
		temp_bos_support = 10
	}
	
		
	clamp_temp_variable = {
		var = temp_bos_support
		min = 1
		max = 11
	}
	add_to_temp_variable = {
		temp_bos_support = 1
	}
	set_variable = { current_bosnian_discontent_frame = temp_bos_support }

}
