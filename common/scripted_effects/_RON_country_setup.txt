RON_handle_party_names = {
		if = { limit = { has_start_date > 1932.12.31 has_start_date < 1949.1.1 } RON_handle_party_names_WW2 = yes }
		if = { limit = { has_start_date > 1948.12.31 has_start_date < 2000.1.1 } RON_handle_party_names_CW = yes }
		if = { limit = { has_start_date > 1999.12.31 } RON_handle_party_names_MD = yes }
}

RON_handle_party_names_event = {
if = {
		limit = {
			RON_party_names_change = yes
		}
		if = { limit = { date > 1932.12.31 date < 1949.1.1 } RON_handle_party_names_WW2 = yes }
		if = { limit = { date > 1948.12.31 date < 2000.1.1 } RON_handle_party_names_CW = yes }
		if = { limit = { date > 1999.12.31 } RON_handle_party_names_MD = yes }
	}
}
