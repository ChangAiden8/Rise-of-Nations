increase_famine = {
	if = {
		limit = {
			has_idea = famine_2
		}
		remove_ideas = famine_2
		add_ideas = famine_3
	}
	else_if = {
		limit = {
			has_idea = famine_1
		}
		add_ideas = famine_2
		remove_ideas = famine_1
	}
	else_if = {
		limit = {
			has_idea = famine_0
		}
		add_ideas = famine_1
		remove_ideas = famine_0
	}

}


tiny_army_desertion = {
	custom_effect_tooltip = tiny_army_desertion_tt
	hidden_effect = {
		random_country = {
			limit = {
				AND = {
					country_exists = THIS
					is_ai = yes
					
					OR = {
						tag = MCO
						tag = ADR
						tag = LIE
						tag = DEL
						tag = VCT
						tag = LUX
						tag = LON
						tag = SMR
						tag = TGR
					}
				}
				
			}
			ROOT = {
				transfer_units_fraction = {
					target = PREV
					size = 0.025
					stockpile_ratio = 0.025
					army_ratio = 0.025
					navy_ratio = 0.025
					air_ratio = 0.025
			
			
				}
				annex_country = { target = PREV transfer_troops = no}
				release = PREV
			}
		}
		
		
	}
}
